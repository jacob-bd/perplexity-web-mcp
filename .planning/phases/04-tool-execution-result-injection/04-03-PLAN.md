```python
from fastapi.testclient import TestClient
from unittest.mock import patch, MagicMock
import pytest

@pytest.fixture
def client():
    from perplexity_web_mcp.api.server import app
    return TestClient(app)

@pytest.fixture
def search_tool():
    return {
        "name": "search",
        "description": "Search for information",
        "input_schema": {
            "type": "object",
            "properties": {"query": {"type": "string"}},
            "required": ["query"]
        }
    }

def test_full_tool_calling_cycle(client, search_tool):
    # Turn 1: Initial query with tools
    with patch('perplexity_web_mcp.api.server.PerplexityClient') as mock_client:
        mock_instance = MagicMock()
        mock_instance.ask.return_value = {
            "answer": "I'll search for that. search('Python tutorials')",
            "citations": []
        }
        mock_client.return_value = mock_instance

        response = client.post(
            "/v1/messages",
            json={
                "model": "claude-3-5-sonnet",
                "messages": [{"role": "user", "content": "Find Python tutorials"}],
                "tools": [search_tool],
                "stream": False
            },
            headers={"x-api-key": "test-key", "anthropic-version": "2023-06-01"}
        )

        assert response.status_code == 200
        data = response.json()
        # Assert tool_use block present
        assert any(block.get("type") == "tool_use" for block in data["content"])
        tool_use_id = next(
            block["id"] for block in data["content"]
            if block.get("type") == "tool_use"
        )

    # Turn 2: Send tool result and continue
    with patch('perplexity_web_mcp.api.server.PerplexityClient') as mock_client:
        mock_instance = MagicMock()
        mock_instance.ask.return_value = {
            "answer": "Based on the search results, here are the Python tutorials...",
            "citations": []
        }
        mock_client.return_value = mock_instance

        response = client.post(
            "/v1/messages",
            json={
                "model": "claude-3-5-sonnet",
                "messages": [
                    {"role": "user", "content": "Find Python tutorials"},
                    {
                        "role": "assistant",
                        "content": [{
                            "type": "tool_use",
                            "id": tool_use_id,
                            "name": "search",
                            "input": {"query": "Python tutorials"}
                        }]
                    },
                    {
                        "role": "user",
                        "content": [{
                            "type": "tool_result",
                            "tool_use_id": tool_use_id,
                            "content": "Found comprehensive Python tutorials at python.org"
                        }]
                    }
                ],
                "tools": [search_tool],
                "stream": False
            },
            headers={"x-api-key": "test-key", "anthropic-version": "2023-06-01"}
        )

        assert response.status_code == 200
        # Verify tool result was injected (check via prompt or response content)
        # The prompt sent to Perplexity should contain the tool result
        # We can verify this via logging or by checking the mock call arguments
        call_args = mock_instance.ask.call_args
        assert "TOOL RESULTS:" in str(call_args) or "Found comprehensive Python tutorials" in str(call_args)
```
